基础类型

数字、字符串和布尔值


// 基础类型示例
const numberType: number = 42;
const stringType: string = "TypeScript";
const booleanType: boolean = true;

// 数组
const numberArray: number[] = [1, 2, 3];
const stringArray: Array = ["TypeScript", "JavaScript"];

// 元组
const tuple: [string, number] = ["age", 25];

// 枚举
enum Direction {
    Up = "UP",
    Down = "DOWN",
    Left = "LEFT",
    Right = "RIGHT"
}

接口和类型

接口定义


interface User {
    id: number;
    name: string;
    age: number;
    email?: string;     // 可选属性
    readonly createdAt: Date;  // 只读属性
}

// 接口继承
interface Employee extends User {
    salary: number;
    department: string;
}

泛型

泛型基础


// 泛型函数
function identity(arg: T): T {
    return arg;
}

// 泛型接口
interface GenericResponse {
    data: T;
    status: number;
    message: string;
}

// 泛型类
class GenericBox {
    private content: T;

    constructor(value: T) {
        this.content = value;
    }

    getValue(): T {
        return this.content;
    }
}

装饰器

类装饰器示例


function Logger(logString: string) {
    return function(constructor: Function) {
        console.log(logString);
        console.log(constructor);
    };
}

@Logger("创建产品...")
class Product {
    constructor(
        private title: string,
        private price: number
    ) {}
}

Interface vs Type

区别示例


// Interface 可以被合并
interface Animal {
    name: string;
}
interface Animal {
    age: number;
}

// Type 不能被合并
type Dog = {
    name: string;
}
// Error: Duplicate identifier 'Dog'
// type Dog = {
//     age: number;
// }

// Interface 使用 extends
interface Employee extends User {
    salary: number;
}

// Type 使用交叉类型
type Worker = User & {
    salary: number;
}
